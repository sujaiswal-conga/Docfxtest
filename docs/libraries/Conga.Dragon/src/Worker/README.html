Conga.Dragon.Worker.Extensions

This module contains extensions to simplify your worker project in your microservice that used Dragon. It provides an abstract class called DragonWorkerStartup that does most of the plumbing you need in a startup class. Your startup class in worker projects should derive from this class.

WorkerConfiguration

This class contains configuration parameters of your worker app.

/// <summary>
/// Contains all the configuration settings of the worker.
/// </summary>
public class WorkerConfiguration
{
	/// <summary>
	/// True to enable dragon.
	/// </summary>
	public bool UnleashDragon { get; set; } = true;

	/// <summary>
	/// Dragon configuration.
	/// </summary>
	public DragonConfig? DragonConfig { get; set; }

	/// <summary>
	/// Dragon configuration.
	/// </summary>
	public DragonStartupParams? DragonStartupParams { get; set; }
}

Example

public class Startup : DragonWorkerStartup
{
    public Startup(IConfiguration configuration, IWebHostEnvironment environment)
    {
        Init(configuration, new WorkerConfiguration
        {
            DragonConfig = new DragonConfig
            {
                ConfigureSubscriber = true
            },
            DragonStartupParams = new DragonStartupParams(new MessagingSetupOptions
            {
                Assemblies = new Assembly[] 
                { 
                    typeof(DocumentWorkflowRequestMessage).Assembly,
                    typeof(DocumentWorkflowRequestMessageHandler).Assembly
                }
            })
        });
    }

    protected override void ConfigureAppServices(IServiceCollection services)
    {
        base.ConfigureAppServices(services);
        services.ConfigureWorkerServices(Configuration);
    }

    protected override void ConfigureAppContainer(IServiceCollection services)
    {
        base.ConfigureAppContainer(services);
        ObjectDBClientRegistration.AddObjectDBClientForWorker(services, Configuration);
        UserManagementClientRegistration.AddUserManagementClientForWorker(services, Configuration);
    }
}
